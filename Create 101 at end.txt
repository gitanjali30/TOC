#include<bits/stdc++.h>
using namespace std;

bool isValid(string s)
{
    int n=s.size();
    if(s[n-1]=='1'&&s[n-2]=='0'&&s[n-3]=='1')
            return true;
    return false;
}
int main()
{
    string name;
    cin>>name;
    if(isValid(name))
        cout<<"String Accepted";
    else
        cout<<"String not Accepted";
        }
